// Várjuk meg, amíg a DOM teljesen betöltődik
document.addEventListener('DOMContentLoaded', () => {

    // --- 1. Feladat: Gomb kattintás ---
    const alertBtn = document.getElementById('alertBtn');
    const messageOutput = document.getElementById('messageOutput');

    if (alertBtn) {
        alertBtn.addEventListener('click', () => {
            messageOutput.textContent = "Sikeresen megnyomtad a gombot! 🎉";
            // 3 másodperc után eltűnik az üzenet
            setTimeout(() => { 
                messageOutput.textContent = ""; 
            }, 3000);
        });
    }

    // --- 2. Feladat: Szöveg megváltoztatása ---
    const changeTextBtn = document.getElementById('changeTextBtn');
    const targetParagraph = document.getElementById('targetParagraph');

    if (changeTextBtn) {
        changeTextBtn.addEventListener('click', () => {
            targetParagraph.textContent = "A szöveg sikeresen megváltozott! ✨";
            targetParagraph.classList.add('text-highlight');
        });
    }

    // --- 3. Feladat: Háttérszín változtatása ---
    const bgChangeBtn = document.getElementById('bgChangeBtn');

    if (bgChangeBtn) {
        bgChangeBtn.addEventListener('click', () => {
            // Véletlenszerű RGB szín generálása
            const r = Math.floor(Math.random() * 256);
            const g = Math.floor(Math.random() * 256);
            const b = Math.floor(Math.random() * 256);
            
            // Beállítjuk a hátteret egy kis átlátszósággal (0.3), hogy ne legyen túl erős
            document.body.style.backgroundColor = `rgb(${r}, ${g}, ${b}, 0.3)`;
        });
    }

    // --- 4. Feladat: Kép váltó galéria ---
    const nextImgBtn = document.getElementById('nextImgBtn');
    const galleryImg = document.getElementById('galleryImg');

    // Példa képek listája (Lorem Picsum ID-k)
    const imageIds = [10, 25, 32, 48, 55, 77];
    let currentIndex = 0;

    if (nextImgBtn && galleryImg) {
        nextImgBtn.addEventListener('click', () => {
            // Index növelése (visszaugrik az elejére, ha a végére ér)
            currentIndex = (currentIndex + 1) % imageIds.length;
            
            // Elhalványítás effektus
            galleryImg.style.opacity = '0';
            
            // Rövid várakozás után képcsere és visszahozás
            setTimeout(() => {
                galleryImg.src = `https://picsum.photos/id/${imageIds[currentIndex]}/400/200`;
                galleryImg.style.opacity = '1';
            }, 250);
        });
    }
});
